EXAMPLE USAGE
═══════════════════════════════════════════════════════════

const { connectDB, getDB } = require('./functions/mongodb');
const mongoHelpers = require('./functions/mongoHelpers');

// Initialize
await connectDB();

// Create user
const user = await mongoHelpers.createUser({
  email: 'john@example.com',
  password: 'hashed_password',
  firstName: 'John',
  lastName: 'Doe'
});

// Generate OTP
const { otp } = await mongoHelpers.generateOTP('john@example.com');

// Verify OTP
await mongoHelpers.verifyOTP('john@example.com', otp);
